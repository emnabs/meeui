按需引入组件（Element UI 风格）

可通过 babel-plugin-component 实现按需引入

// babel.config.js 配置
plugins: [
  [
    'component',
    {
      libraryName: 'meeui',
      styleLibraryName: 'theme'
    }
  ]
]
      
// 组件使用示例
import Vue from 'vue';
import { Button, Cell } from 'meeui';

Vue.use(Button);
Vue.use(Cell);
